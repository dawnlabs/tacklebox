# Tackle Box
> Collection of React user-experience containers for common interactions

## TempValue

import {TempValue} from '../src'

<Playground>
  <TempValue initialValue="Mike">
    {({ hasChanged, value, onInputChange, submit, reset }) => (
      <form onSubmit={e => e.preventDefault()}>
        <input value={value || ''} onChange={onInputChange} />
        <button
          style={{ cursor: hasChanged ? 'pointer' : 'not-allowed'}}
          disabled={!hasChanged}
          onClick={submit}
        >
            Submit
        </button>
        <button
          style={{ cursor: hasChanged ? 'pointer' : 'not-allowed'}}
          disabled={!hasChanged}
          onClick={reset}
        >
          Cancel
        </button>
      </form>
    )}
  </TempValue>
</Playground>
